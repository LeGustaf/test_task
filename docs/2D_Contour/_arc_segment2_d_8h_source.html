82 void calculateFromEndpoints() {
83 double dx = pointB.getX() - pointA.getX();
84 double dy = pointB.getY() - pointA.getY();
85 double chordLen = std::sqrt(dx * dx + dy * dy);
86 double minRadius = chordLen / 2;
87 if (minRadius > radius) {
88 std::cerr << "Radius too small, clamping to " << (minRadius * 1.1) << std::endl;
89 radius = minRadius * 1.1;
90 }
91
92 double mx = (pointB.getX() + pointA.getX()) / 2;
93 double my = (pointB.getY() + pointA.getY()) / 2;
94 double h = std::sqrt(radius * radius - (chordLen * chordLen) / 4);
95
96 double perp_dx = -dy;
97 double perp_dy = dx;
98 double mag = std::sqrt(perp_dx * perp_dx + perp_dy * perp_dy);
99 perp_dx /= mag;
100 perp_dy /= mag;
101
102 // Adjust direction
103 if (!clockwise) {
104 perp_dx = -perp_dx;
105 perp_dy = -perp_dy;
106 }
107
108 double cx = mx + h * perp_dx;
109 double cy = my + h * perp_dy;
110
111 center = MyPoint(cx, cy);
112 // Ensure angle is in range [-PI, PI] if needed later for arc traversal
113 startAngle = atan2(pointA.getY() - cy, pointA.getX() - cx);
114 endAngle = atan2(pointB.getY() - cy, pointB.getX() - cx);

116 length();
